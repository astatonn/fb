# Configuração para froeneradvocacia.com
# Adicione esta seção no seu nginx.conf

    # Configuração para froeneradvocacia.com
    server {
        listen 443 ssl http2;
        server_name froeneradvocacia.com www.froeneradvocacia.com;

        ssl_certificate /etc/letsencrypt/live/froeneradvocacia.com/fullchain.pem;
        ssl_certificate_key /etc/letsencrypt/live/froeneradvocacia.com/privkey.pem;

        # Headers de segurança
        add_header Strict-Transport-Security "max-age=31536000; includeSubDomains" always;
        add_header X-Frame-Options "SAMEORIGIN" always;
        add_header X-Content-Type-Options "nosniff" always;
        add_header Referrer-Policy "no-referrer-when-downgrade" always;

        # Cache para arquivos estáticos
        location ~* \.(jpg|jpeg|png|gif|ico|css|js|svg|woff|woff2|ttf|eot|webp)$ {
            proxy_pass http://froener-advocacia-prod:80;
            proxy_set_header Host $host;
            proxy_set_header X-Forwarded-Proto $scheme;
            expires 7d;
            add_header Cache-Control "public, max-age=604800";
        }

        # Configuração principal para todas as requisições
        location / {
            proxy_pass http://froener-advocacia-prod:80;
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto $scheme;
        }

        location ~ /\.ht {
            deny all;
        }

        error_page 404 https://astatonn.com;
    }

# IMPORTANTE: Após adicionar esta configuração:
# 1. Gerar certificado SSL: certbot --nginx -d froeneradvocacia.com -d www.froeneradvocacia.com
# 2. Recarregar nginx: sudo systemctl reload nginx
# 3. Alterar docker-compose.prod.yml para remover a porta 8080 (não expor externamente)
